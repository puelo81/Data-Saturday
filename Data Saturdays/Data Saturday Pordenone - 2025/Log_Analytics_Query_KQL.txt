//This query Combines Log Analytics Value with Execution Metrics MaxPeakMem to include Mem metrics on Queries – //Includes also an anomaly detection calculation based on Peak Memory Elapsed. Can be Executed on Log Analytics -//Log tab on Portal or .. either embedded in an Excel File or PBIX
let LookbackPeriod = 1d;  // Adjust lookback period if needed
let AnomalyThreshold = 3;  // Adjust threshold for anomaly detection (z-score)
//let TargetUser = "users@mail.com";  // Replace with the specific user to filter
let executionMetrics = 
    PowerBIDatasetsWorkspace
    | where TimeGenerated >= ago(LookbackPeriod)
    | where OperationName == "ExecutionMetrics"
    | project 
        XmlaRequestId, 
        OperationName,
        EventText,
        MaxPeakMemoryGB = round(todouble(parse_json(EventText).approximatePeakMemConsumptionKB) / 1048576,2);  // Convert Kbytes to GB  
let MemStats = 
    executionMetrics
    | summarize AvgPeakMem = avg(MaxPeakMemoryGB), StdPeakMem = stdev(MaxPeakMemoryGB);
PowerBIDatasetsWorkspace
    | where TimeGenerated >= ago(LookbackPeriod)
    | where OperationName in ("QueryEnd", "QueryFailed")
    | project
        XmlaRequestId,
        TimeGenerated,
        PowerBIWorkspaceName,
        ArtifactName,
        OperationName,
        OperationDetailName,
        EventText,
        CpuTimeMs,
        DurationMs,
        ExecutingUser,
        DurationSec = DurationMs / 1000.0
        | join kind=leftouter executionMetrics on XmlaRequestId
        | extend MaxPeakMemoryGB, // Convert bytes to GB and round
